 [수강신청 데이터베이스]
학생(학번, 이름, 전공, 학년)
수강(과목코드, 학번, 수강학기, 성적)
과목(과목코드, 과목이름, 강의실, 요일, 담당교수)

(1)학생 수강 전체 현황 뷰를 작성하시오.
(학생정보+수강 과목+성적을 한 분에 확인가능하도록 작성)
CREATE VIEW v_student_enroll_all AS
SELECT s.학번, s.이름, s.전공, s.학년,
       c.과목코드, c.과목이름, c.강의실, c.요일, c.담당교수,
       e.수강학기, e.성적
FROM 학생 s
JOIN 수강 e ON s.학번 = e.학번
JOIN 과목 c ON e.과목코드 = c.과목코드;

(2)각 학생별 평균 성적 뷰를 작성하시오.
(각 학생의 전체 수강 과목 평균 성적 계산 뷰)
CREATE VIEW v_student_avg AS
SELECT 학번,
       AVG(성적) AS 평균성적
FROM 수강
GROUP BY 학번;

(3)과목별 수강 인원 및 평균 성적 뷰를 작성하시오.
(각 과목의 수강 인원과 평균, 최고,최저 성적 통계 뷰)
CREATE VIEW v_subject_stats AS
SELECT 과목코드,
       COUNT(학번) AS 수강인원,
       AVG(성적) AS 평균성적,
       MAX(성적) AS 최고성적,
       MIN(성적) AS 최저성적
FROM 수강
GROUP BY 과목코드;

(4)성적 우수 학생 뷰를 작성하시오.(평균 90점 이상)
(평균 성적이 90점 이상이 우수 학생만 조회하는 뷰)
CREATE VIEW v_top_student AS
SELECT 학번, AVG(성적) AS 평균성적
FROM 수강
GROUP BY 학번
HAVING AVG(성적) >= 90;

(5)전공 별 수강 통계 뷰를 작성하시오.
(전공별 총 수강 건수와 전공 평균 성적 비교 뷰)
CREATE VIEW v_major_stats AS
SELECT s.전공,
       COUNT(e.과목코드) AS 총수강건수,
       AVG(e.성적) AS 평균성적
FROM 학생 s
JOIN 수강 e ON s.학번 = e.학번
GROUP BY s.전공;

(6)학기별 수강 현황 뷰를 작성하시오.
CREATE VIEW v_semester_stats AS
SELECT 수강학기,
       COUNT(*) AS 수강건수,
       AVG(성적) AS 평균성적
FROM 수강
GROUP BY 수강학기;

(7)미수강 학생 뷰를 작성하시오.
CREATE VIEW v_no_enroll_student AS
SELECT 학번, 이름, 전공, 학년
FROM 학생
WHERE 학번 NOT IN (SELECT 학번 FROM 수강 WHERE 학번 IS NOT NULL);

(8)담당 교수별 강의 및 수강 현황 뷰를 작성하시오.
  과목별 담당 과목 수와 총 수강 학생수를 집계하는 뷰)
CREATE VIEW v_professor_stats AS
SELECT c.담당교수,
       COUNT(DISTINCT c.과목코드) AS 과목수,
       COUNT(e.학번) AS 총수강학생수
FROM 과목 c
LEFT JOIN 수강 e ON c.과목코드 = e.과목코드
GROUP BY c.담당교수;

(9)학년별 수강 과목 수 및 평균 성적 뷰를 작성하시오.
(학년에 따른 수강 부담과 성적 추이를 파악하는 뷰 작성)
CREATE VIEW v_grade_stats AS
SELECT s.학년,
       COUNT(e.과목코드) AS 수강과목수,
       AVG(e.성적) AS 평균성적
FROM 학생 s
JOIN 수강 e ON s.학번 = e.학번
GROUP BY s.학년;

(10)성적 미입력(NULL) 수강 내역 뷰를 작성하시오.
(성적이 아직 입력되지 않은 수강 내역 관리용이다.)
CREATE VIEW v_null_grade AS
SELECT *
FROM 수강
WHERE 성적 IS NULL;


 테이블 구조 확인
극장(극장번호, 극장이름, 위치)
상영관(극장번호, 상영관번호, 영화제목, 가격, 좌석수)
예약(극장번호, 상영관번호, 고객번호, 좌석번호, 날짜)
고객(고객번호, 이름, 주소)
 뷰 생성 질의어 15개
1. 전체 상영 정보 통합 뷰를 작성하시오.
(극장 + 상영관 + 영화 정보를 한눈에 확인 가능하도록 작성)
CREATE VIEW v_all_screen_info AS
SELECT t.극장번호, t.극장이름, t.위치,
       s.상영관번호, s.영화제목, s.가격, s.좌석수
FROM 극장 t
JOIN 상영관 s ON t.극장번호 = s.극장번호;

2 .예약 전체 현황 뷰를 작성하시오
(고객이 어떤 극장·상영관·영화를 예약했는지 통합 조회하도록 작성)
CREATE VIEW v_reservation_all AS
SELECT c.고객번호, c.이름, c.주소,
       r.극장번호, r.상영관번호, r.좌석번호, r.날짜,
       s.영화제목, s.가격
FROM 고객 c
JOIN 예약 r ON c.고객번호 = r.고객번호
JOIN 상영관 s ON r.극장번호 = s.극장번호
AND r.상영관번호 = s.상영관번호;

3. 극장별 상영 영화 목록 뷰를 작성하시오
  (각 극장에서 상영 중인 영화와 가격, 좌석수 조회)
CREATE VIEW v_theater_movie_list AS
SELECT t.극장이름, t.위치,
       s.영화제목, s.가격, s.좌석수
FROM 극장 t
JOIN 상영관 s ON t.극장번호 = s.극장번호;

4. 고객별 예약 내역 뷰를 작성하시오
  (고객별 예약한 영화 목록과 날짜, 금액 조회가능한 뷰)
CREATE VIEW v_customer_reservation AS
SELECT c.이름, c.주소,
       s.영화제목,
       r.날짜,
       s.가격
FROM 고객 c
JOIN 예약 r ON c.고객번호 = r.고객번호
JOIN 상영관 s ON r.극장번호 = s.극장번호
AND r.상영관번호 = s.상영관번호;

5. 상영관별 예약 인원 및 잔여 좌석 뷰를 작성하시오
  (상영관의 총 좌석 대비 예약 현황과 잔여 좌석 계산)
CREATE VIEW v_screen_seat_status AS
SELECT s.극장번호, s.상영관번호, s.영화제목,
       s.좌석수,
       COUNT(r.좌석번호) AS 예약수,
       (s.좌석수 - COUNT(r.좌석번호)) AS 잔여좌석
FROM 상영관 s
LEFT JOIN 예약 r
ON s.극장번호 = r.극장번호
AND s.상영관번호 = r.상영관번호
GROUP BY s.극장번호, s.상영관번호, s.영화제목, s.좌석수;

6. 극장별 총 예약 수 및 매출 뷰를 작성하시오
  (극장별 총 예약 건수와 누적 매출 집계를 볼 수 있는 뷰)
CREATE VIEW v_theater_sales AS
SELECT t.극장이름,
       COUNT(r.좌석번호) AS 예약건수,
       SUM(s.가격) AS 매출
FROM 극장 t
JOIN 상영관 s ON t.극장번호 = s.극장번호
JOIN 예약 r ON s.극장번호 = r.극장번호
AND s.상영관번호 = r.상영관번호
GROUP BY t.극장이름;

7. 영화별 예약 현황 뷰를 작성하시오
  (영화 제목별 총 예약 인원과 총 매출 집계 뷰)
CREATE VIEW v_movie_stats AS
SELECT s.영화제목,
       COUNT(r.좌석번호) AS 예약수,
       SUM(s.가격) AS 매출
FROM 상영관 s
JOIN 예약 r ON s.극장번호 = r.극장번호
AND s.상영관번호 = r.상영관번호
GROUP BY s.영화제목;
      
8. 날짜별 예약 현황 뷰를 작성하시오
  (날짜별 예약 건수와 해당 날짜 매출 조회 가능한 뷰 작성)
CREATE VIEW v_date_stats AS
SELECT r.날짜,
       COUNT(*) AS 예약건수,
       SUM(s.가격) AS 매출
FROM 예약 r
JOIN 상영관 s ON r.극장번호 = s.극장번호
AND r.상영관번호 = s.상영관번호
GROUP BY r.날짜;

9. 예약 이력이 없는 고객 뷰
(한 번도 예약한 적 없는 고객 조회하는 뷰 작서)
CREATE VIEW v_no_reservation_customer AS
SELECT *
FROM 고객
WHERE 고객번호 NOT IN (
    SELECT 고객번호 FROM 예약 WHERE 고객번호 IS NOT NULL
);

10. 고객별 총 예약 횟수 및 결제 금액 뷰
(고객의 예약 활동 통계) (충성 고객 분석용)
CREATE VIEW v_customer_stats AS
SELECT c.이름,
       COUNT(r.좌석번호) AS 예약횟수,
       SUM(s.가격) AS 결제금액
FROM 고객 c
JOIN 예약 r ON c.고객번호 = r.고객번호
JOIN 상영관 s ON r.극장번호 = s.극장번호
AND r.상영관번호 = s.상영관번호
GROUP BY c.이름;

11. 가격이 가장 비싼 상영관 뷰를 작성하시오
   (티켓 가격 기준 상위 상영관 목록 조회하는 뷰 작성)
CREATE VIEW v_max_price_screen AS
SELECT *
FROM 상영관
WHERE 가격 = (SELECT MAX(가격) FROM 상영관);

12. 위치별 극장 및 상영 현황 뷰
 (지역(위치)별로 운영 중인 극장과 상영 영화 수 집계하는 뷰 작성)
CREATE VIEW v_location_stats AS
SELECT t.위치,
       COUNT(DISTINCT t.극장번호) AS 극장수,
       COUNT(s.상영관번호) AS 상영관수
FROM 극장 t
JOIN 상영관 s ON t.극장번호 = s.극장번호
GROUP BY t.위치;

13. 만석 상영관 뷰를 작성하시오
  (예약 좌석 수가 전체 좌석 수와 동일한 매진 상영관 조회)
CREATE VIEW v_full_screen AS
SELECT s.극장번호, s.상영관번호, s.영화제목
FROM 상영관 s
JOIN 예약 r ON s.극장번호 = r.극장번호
AND s.상영관번호 = r.상영관번호
GROUP BY s.극장번호, s.상영관번호, s.영화제목, s.좌석수
HAVING COUNT(r.좌석번호) = s.좌석수;

14. 특정 날짜 예약 고객 상세 뷰를 작성하시오
( 날짜별 예약 고객 정보와 관람 영화 상세 조회)
CREATE VIEW v_date_customer AS
SELECT c.이름, c.주소,
       s.영화제목,
       r.날짜
FROM 고객 c
JOIN 예약 r ON c.고객번호 = r.고객번호
JOIN 상영관 s ON r.극장번호 = s.극장번호
AND r.상영관번호 = s.상영관번호;

15. 예약 없는 상영관 뷰
개설되었지만 예약이 한 건도 없는 상영관 조회
CREATE VIEW v_no_reservation_screen AS
SELECT s.*
FROM 상영관 s
LEFT JOIN 예약 r
ON s.극장번호 = r.극장번호
AND s.상영관번호 = r.상영관번호
WHERE r.상영관번호 IS NULL;


[판매원 데이터베이스]
판매원 테이블
Salesperson1(name, age, salary)
고객 테이블
Customer1(name, city, industrytype)
주문 테이블
Orders1(order_id, custname, salesperson, amount)-- 컬럼 설명
name         -- 판매원/고객 이름
age          -- 판매원 나이
salary       -- 판매원 급여
city         -- 고객 거주 도시
industrytype -- 고객 직업
number       -- 주문번호
custname     -- 주문 고객 이름
salesperson  -- 담당 판매원 이름
amount       -- 주문 금액

 뷰 생성 질의어 15개
1. 전체 주문 통합 현황 뷰를 작성하시오
(주문 + 고객 + 영업사원 정보를 한눈에 통합 조회 뷰)
CREATE VIEW v_order_full AS
SELECT o.order_id, o.amount,
       c.name AS 고객명, c.city, c.industrytype,
       s.name AS 영업사원명, s.age, s.salary
FROM Orders1 o
JOIN Customer1 c ON o.custname = c.name
JOIN Salesperson1 s ON o.salesperson = s.name;

 
2. 영업사원별 총 주문 금액 및 건수 뷰를 작성하시오
(영업사원별 실적(주문 건수, 총 매출) 집계)
CREATE VIEW v_salesperson_sales AS
SELECT salesperson AS 영업사원,
       COUNT(order_id) AS 주문건수,
       SUM(amount) AS 총매출
FROM Orders1
GROUP BY salesperson;

3. 고객별 총 주문 금액 뷰를 작성하시오
(고객별 누적 주문 횟수와 총 구매 금액 집계 포함된 뷰)
CREATE VIEW v_customer_sales AS
SELECT custname AS 고객명,
       COUNT(order_id) AS 주문횟수,
       SUM(amount) AS 총구매금액
FROM Orders1
GROUP BY custname;

4. 도시별 주문 현황 뷰를 작성하시오
(도시(city)별 총 주문 건수와 매출 집계 포함된 뷰)
CREATE VIEW v_city_sales AS
SELECT c.city,
       COUNT(o.order_id) AS 주문건수,
       SUM(o.amount) AS 총매출
FROM Customer1 c
JOIN Orders1 o ON c.name = o.custname
GROUP BY c.city;

5. 업종별 주문 통계 뷰를 작성하시오
  (산업 유형(industrytype)별 주문 건수 및 총 매출 비교 뷰)
CREATE VIEW v_industry_sales AS
SELECT c.industrytype,
       COUNT(o.order_id) AS 주문건수,
       SUM(o.amount) AS 총매출
FROM Customer1 c
JOIN Orders1 o ON c.name = o.custname
GROUP BY c.industrytype;

6. 고액 주문 뷰 (amount 상위)를 작성하시오
  (주문 금액이 평균 이상인 고액 주문만 조회하는 뷰)
CREATE VIEW v_high_order AS
SELECT *
FROM Orders1
WHERE amount >= (SELECT AVG(amount) FROM Orders1);

7. 주문 실적 없는 영업사원 뷰를 작성하시오
   (한 건도 주문을 처리하지 않은 영업사원 조회 뷰)
CREATE VIEW v_no_order_salesperson AS
SELECT *
FROM Salesperson1
WHERE name NOT IN (
    SELECT salesperson FROM Orders1 WHERE salesperson IS NOT NULL
);

8. 주문 이력 없는 고객 뷰를 작성하시오
  (등록은 되어 있지만 한 번도 주문하지 않은 고객 조회 뷰)
CREATE VIEW v_no_order_customer AS
SELECT *
FROM Customer1
WHERE name NOT IN (
    SELECT custname FROM Orders1 WHERE custname IS NOT NULL
);

9. 영업사원 급여 등급 뷰를 작성하시오
  (급여 구간별로 영업사원을 분류하는 뷰)
CREATE VIEW v_salary_grade AS
SELECT name, salary,
       CASE
           WHEN salary >= 15000 THEN 'A'
           WHEN salary >= 10000 THEN 'B'
           ELSE 'C'
       END AS 등급
FROM Salesperson1;

10. 영업사원별 담당 고객 목록 뷰를 작성하시오
  (각 영업사원이 주문을 처리한 고객 목록(중복 제거) 생성 뷰)
CREATE VIEW v_sales_customer AS
SELECT DISTINCT salesperson AS 영업사원,
                custname AS 고객명
FROM Orders1;

11. 최고 실적 영업사원 뷰를 작성하시오
  (총 주문 금액이 가장 높은 영업사원 조회하는 뷰)
CREATE VIEW v_top_salesperson AS
SELECT salesperson,
       SUM(amount) AS 총매출
FROM Orders1
GROUP BY salesperson
HAVING SUM(amount) = (
    SELECT MAX(total)
    FROM (
        SELECT SUM(amount) AS total
        FROM Orders1
        GROUP BY salesperson
    ) t
);

  
12. 도시별 영업사원 활동 현황 뷰를 작성하시오
  (어느 도시의 고객에게 얼마나 판매했는지 영업사원 기준 집계 뷰)
CREATE VIEW v_city_salesperson AS

       o.salesperson,
       SUM(o.amount) AS 매출

GROUP BY c.city, o.salesperson;

13. 영업사원 급여 대비 매출 효율 뷰를 작성하시오
 (영업사원 급여 대비 총 매출 비율로 효율성 측정 뷰)
CREATE VIEW v_efficiency AS
SELECT s.name,
       s.salary,
       SUM(o.amount) AS 총매출,
       SUM(o.amount) * 1.0 / s.salary AS 효율

14. 업종별 담당 영업사원 현황 뷰를 작성하시오
   (산업 유형별로 어떤 영업사원이 활동하는지 파악)
CREATE VIEW v_industry_sp AS

15. 주문 금액 구간별 분류 뷰를 작성하시오

CREATE VIEW v_amount_grade AS

  테이블 구조 확인 
  Passenger(pid,  pname,  pgender, pcity)
  Agency(aid,  aname,  acity)
  Flight(fid,  fdate,  time,  src,  dest)
  Booking(pid,  aid,  fid,  fdate)
Booking.pid  → Passenger.pid 참조
Booking.aid  → Agency.aid 참조
Booking.fid  → Flight.fid 참조
 뷰 생성 질의어 15개
1. 전체 예약 통합 현황 뷰를 작성하시오
  (승객 + 여행사 + 항공편 정보를 한눈에 통합 조회 뷰)
CREATE VIEW v_booking_full AS
SELECT b.pid, b.aid, b.fid, b.fdate,
       p.pname, p.pgender, p.pcity,
       a.aname, a.acity,
       f.src, f.dest, f.time
FROM Booking b
JOIN Passenger p ON b.pid = p.pid
JOIN Agency a ON b.aid = a.aid
JOIN Flight f ON b.fid = f.fid;

2. 승객별 예약 내역 뷰를 작성하시오
   (각 승객이 예약한 항공편과 여행사 정보 조회 뷰)
CREATE VIEW v_passenger_booking AS
SELECT p.pid, p.pname,
       a.aname,
       f.fid, f.src, f.dest, f.fdate
FROM Passenger p
JOIN Booking b ON p.pid = b.pid
JOIN Agency a ON b.aid = a.aid
JOIN Flight f ON b.fid = f.fid;

3. 여행사별 예약 건수 및 실적 뷰를 작성하시오
   (각 여행사의 총 예약 건수와 취급 항공편 수 집계 뷰)
CREATE VIEW v_agency_stats AS
SELECT a.aname,
       COUNT(b.fid) AS 예약건수,
       COUNT(DISTINCT b.fid) AS 취급항공편수
FROM Agency a
LEFT JOIN Booking b ON a.aid = b.aid
GROUP BY a.aname;

4. 항공편별 예약 승객 수 뷰를 작성하시오
   (각 항공편의 예약된 승객 수와 출발·도착지 조회 뷰)
CREATE VIEW v_flight_passenger AS
SELECT f.fid, f.src, f.dest,
       COUNT(b.pid) AS 예약승객수
FROM Flight f
LEFT JOIN Booking b ON f.fid = b.fid
GROUP BY f.fid, f.src, f.dest;

5. 출발지·도착지별 노선 통계 뷰를 작성하시오
  (노선(src → dest)별 운항 횟수와 총 예약 건수 집계 뷰)
CREATE VIEW v_route_stats AS
SELECT src, dest,
       COUNT(*) AS 운항횟수
FROM Flight
GROUP BY src, dest;

6. 도시별 승객 예약 현황 뷰를 작성하시오
   (승객 거주 도시별 총 예약 건수 및 이용 현황 집계 뷰)
CREATE VIEW v_city_booking AS
SELECT p.pcity,
       COUNT(b.fid) AS 예약건수
FROM Passenger p
JOIN Booking b ON p.pid = b.pid
GROUP BY p.pcity;

7. 성별 이용 통계 뷰를 작성하시오
   (성별(pgender)에 따른 예약 건수와 이용 노선 수 비교 뷰)
CREATE VIEW v_gender_stats AS
SELECT p.pgender,
       COUNT(b.fid) AS 예약건수,
       COUNT(DISTINCT b.fid) AS 이용노선수
FROM Passenger p
JOIN Booking b ON p.pid = b.pid
GROUP BY p.pgender;

8. 예약 이력 없는 승객 뷰를 작성하시오
   (한 번도 예약하지 않은 승객 조회 (비활성 승객 관리) 뷰)
CREATE VIEW v_no_booking_passenger AS
SELECT *
FROM Passenger
WHERE pid NOT IN (
    SELECT pid FROM Booking WHERE pid IS NOT NULL
);

9. 예약 실적 없는 여행사 뷰를 작성하시오
   (한 건도 예약을 처리하지 않은 여행사 조회 뷰)
CREATE VIEW v_no_booking_agency AS
SELECT *
FROM Agency
WHERE aid NOT IN (
    SELECT aid FROM Booking WHERE aid IS NOT NULL
);

10. 승객별 이용 여행사 목록 뷰를 작성하시오
   (각 승객이 이용한 여행사 목록 (중복 제거) 뷰)
CREATE VIEW v_passenger_agency AS
SELECT DISTINCT p.pname, a.aname
FROM Passenger p
JOIN Booking b ON p.pid = b.pid
JOIN Agency a ON b.aid = a.aid;

11. 동일 출발·도착지 왕복 노선 뷰를 작성하시오
   (왕복 운항이 가능한 노선 쌍(A→B, B→A) 조회 뷰)
CREATE VIEW v_round_trip AS
SELECT f1.src, f1.dest
FROM Flight f1
JOIN Flight f2
ON f1.src = f2.dest AND f1.dest = f2.src;

12. 여행사별 담당 승객 상세 뷰를 작성하시오
   (여행사별로 담당한 승객 정보와 이용 항공편 상세 조회 뷰)
CREATE VIEW v_agency_passenger_detail AS
SELECT a.aname, p.pname,
       f.fid, f.src, f.dest, f.fdate
FROM Agency a
JOIN Booking b ON a.aid = b.aid
JOIN Passenger p ON b.pid = p.pid
JOIN Flight f ON b.fid = f.fid;

13. 날짜별 예약 및 항공편 현황 뷰를 작성하시오
   (날짜별 총 예약 건수와 운항 항공편 수 집계 뷰)
CREATE VIEW v_date_stats AS
SELECT f.fdate,
       COUNT(b.pid) AS 예약건수,
       COUNT(DISTINCT f.fid) AS 항공편수
FROM Flight f
LEFT JOIN Booking b ON f.fid = b.fid
GROUP BY f.fdate;

14. 다중 항공편 이용 승객 뷰를 작성하시오
   (2개 이상의 항공편을 예약한 승객 조회 (자주 이용 고객) 뷰)
CREATE VIEW v_frequent_passenger AS
SELECT pid,
       COUNT(fid) AS 이용항공편수
FROM Booking
GROUP BY pid
HAVING COUNT(fid) >= 2;

15. 승객-여행사 동일 도시 예약 뷰를 작성하시오
   (승객의 거주 도시와 여행사 위치가 같은 예약 조회 뷰)
CREATE VIEW v_same_city_booking AS
SELECT p.pname, p.pcity,
       a.aname, a.acity,
       f.fid
FROM Passenger p
JOIN Booking b ON p.pid = b.pid
JOIN Agency a ON b.aid = a.aid
JOIN Flight f ON b.fid = f.fid
WHERE p.pcity = a.acity;


[기업프로젝트 데이터베이스]
(문제의 컬럼명과 내가 가지고 있는 컬럼명이 다른게있어서 아래와같이 수정)
Employee(empno, name, phoneno, address, sex, position, depno)
Department(depno, deptname, manager)
Project(projno, projname, deptno)
Works(empno, projno, hours_worked)

 [뷰 생성 질의어 ]
1. 직원 전체 정보 통합 뷰를 작성하시오
  (직원 정보와 소속 부서명을 함께 조회하는 뷰)
CREATE VIEW v_emp_full AS
SELECT e.empno, e.name, e.phoneno, e.address, e.sex, e.position,
       d.deptname
FROM Employee e
JOIN Department d ON e.depno = d.depno;

2. 직원별 프로젝트 참여 현황 뷰를 작성하시오
  (직원이 참여 중인 프로젝트명과 투입 시간 조회하는 뷰)
CREATE VIEW v_emp_project AS
SELECT e.empno, e.name,
       p.projname,
       w.hours_worked
FROM Employee e
JOIN Works w ON e.empno = w.empno
JOIN Project p ON w.projno = p.projno;

3. 부서별 직원 수 및 현황 뷰를 작성하시오
  (부서별 소속 직원 수와 관리자 정보 집계 뷰)
CREATE VIEW v_dept_emp AS
SELECT d.depno, d.deptname, d.manager,
       COUNT(e.empno) AS emp_count
FROM Department d
LEFT JOIN Employee e ON d.depno = e.depno
GROUP BY d.depno, d.deptname, d.manager;

4. 프로젝트별 참여 직원 및 총 투입 시간 뷰를 작성하시오
  (각 프로젝트에 참여한 직원 수와 총 근무 시간 집계 뷰)
CREATE VIEW v_project_emp AS
SELECT p.projno, p.projname,
       COUNT(w.empno) AS emp_count,
       SUM(w.hours_worked) AS total_hours
FROM Project p
LEFT JOIN Works w ON p.projno = w.projno
GROUP BY p.projno, p.projname;

5. 직원별 총 근무 시간 뷰를 작성하시오
   (직원이 전체 프로젝트에 투입한 총 시간 집계 뷰)
CREATE VIEW v_emp_hours AS
SELECT e.empno, e.name,
       SUM(w.hours_worked) AS total_hours
FROM Employee e
JOIN Works w ON e.empno = w.empno
GROUP BY e.empno, e.name;

6. 부서별 프로젝트 현황 뷰를 작성하시오
   (각 부서가 담당하는 프로젝트 수와 총 투입 시간 집계 뷰)
CREATE VIEW v_dept_project AS
SELECT d.depno, d.deptname,
       COUNT(p.projno) AS project_count,
       NVL(SUM(w.hours_worked),0) AS total_hours
FROM Department d
LEFT JOIN Project p ON d.depno = p.deptno
LEFT JOIN Works w ON p.projno = w.projno
GROUP BY d.depno, d.deptname;

7. 프로젝트에 참여하지 않는 직원 뷰를 작성하시오
  (어떤 프로젝트에도 투입되지 않은 직원 조회 뷰)
CREATE VIEW v_no_project_emp AS
SELECT e.empno, e.name
FROM Employee e
LEFT JOIN Works w ON e.empno = w.empno
WHERE w.empno IS NULL;

8. 성별 프로젝트 참여 통계 뷰를 작성하시오
   (성별(sex)에 따른 프로젝트 참여 수와 평균 근무 시간 비교 뷰)
CREATE VIEW v_gender_project AS
SELECT e.sex,
       COUNT(w.projno) AS project_count,
       AVG(w.hours_worked) AS avg_hours
FROM Employee e
JOIN Works w ON e.empno = w.empno
GROUP BY e.sex;

9. 관리자(Manager) 직원 상세 뷰를 작성하시오
   (부서 관리자로 지정된 직원의 상세 정보 조회 뷰)
CREATE VIEW v_manager_emp AS
SELECT e.empno, e.name, e.phoneno, e.address, e.sex, e.position
FROM Employee e
JOIN Department d ON e.empno = d.manager;

10. 직책별 프로젝트 참여 현황 뷰를 작성하시오
   (직책(position)별 총 프로젝트 참여 건수와 평균 투입 시간 뷰)
CREATE VIEW v_position_project AS
SELECT e.position,
       COUNT(w.projno) AS project_count,
       AVG(w.hours_worked) AS avg_hours
FROM Employee e
JOIN Works w ON e.empno = w.empno
GROUP BY e.position;

11. 다중 프로젝트 참여 직원 뷰를 작성하시오
   (2개 이상의 프로젝트에 참여 중인 직원 조회 뷰)
CREATE VIEW v_multi_project_emp AS
SELECT e.empno, e.name
FROM Employee e
JOIN Works w ON e.empno = w.empno
GROUP BY e.empno, e.name
HAVING COUNT(w.projno) >= 2;

12. 부서 내 프로젝트 참여 직원 상세 뷰를 작성하시오
(같은 부서 소속 직원이 해당 부서 프로젝트에 참여하는 현황 뷰)
CREATE VIEW v_dept_project_emp AS
SELECT e.empno, e.name, d.deptname, p.projname
FROM Employee e
JOIN Department d ON e.depno = d.depno
JOIN Works w ON e.empno = w.empno
JOIN Project p ON w.projno = p.projno
WHERE e.depno = p.deptno;

13. 타 부서 프로젝트 참여 직원 뷰를 작성하시오
(자신의 소속 부서가 아닌 타 부서 프로젝트에 참여하는 직원 뷰)
CREATE VIEW v_other_dept_emp AS
SELECT e.empno, e.name, e.depno, p.projname
FROM Employee e
JOIN Works w ON e.empno = w.empno
JOIN Project p ON w.projno = p.projno
WHERE e.depno <> p.deptno;

14. 고투입 시간 직원 뷰 (평균 이상 근무)를 작성하시오
(프로젝트 평균 투입 시간보다 많이 근무한 직원 조회 뷰)
CREATE VIEW v_high_hours_emp AS
SELECT e.empno, e.name,
       SUM(w.hours_worked) AS total_hours
FROM Employee e
JOIN Works w ON e.empno = w.empno
GROUP BY e.empno, e.name
HAVING SUM(w.hours_worked) >
       (SELECT AVG(hours_worked) FROM Works);

15. 프로젝트 미배정 부서 뷰를 작성하시오
(담당 프로젝트가 하나도 없는 부서 조회 뷰)
CREATE VIEW v_no_project_dept AS
SELECT d.depno, d.deptname
FROM Department d
LEFT JOIN Project p ON d.depno = p.deptno
WHERE p.projno IS NULL;
